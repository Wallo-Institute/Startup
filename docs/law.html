async function loadLaws() { const apiURL = './laws.json'; const container = document.getElementById('lawContainer'); const searchInput = document.getElementById('lawSearch'); // Show a loading spinner or text while waiting container.innerHTML = '
Consulting the Virginia Code...
'; try { const response = await fetch(apiURL); // Handle the case where the file hasn't been generated yet if (!response.ok) { throw new Error("Law database is currently being updated. Please check back in a few minutes."); } const laws = await response.json(); // Check if the array is actually empty if (laws.length === 0) { container.innerHTML = '
No statutes found at this time.
'; return; } function displayLaws(filteredLaws) { if (filteredLaws.length === 0) { container.innerHTML = '
No matching statutes found.
'; return; } container.innerHTML = filteredLaws.map(law => `
${law.code}
${law.title}

${law.description}

`).join(''); } displayLaws(laws); searchInput.addEventListener('input', (e) => { const term = e.target.value.toLowerCase(); const filtered = laws.filter(l => l.title.toLowerCase().includes(term) || l.code.toLowerCase().includes(term) ); displayLaws(filtered); }); } catch (error) { console.error(error); container.innerHTML = `
${error.message}
`; } } loadLaws();
